void FFT(Complex a[], int n, int inv) {
    // ä½é€†åºç½®æ¢
    for (int i = 0, j = 0; i < n; i++) {
        if (i > j)
            swap(a[i], a[j]);
        for (int k = n >> 1; (j ^= k) < k; k >>= 1)
            ;

            Complex w(1, 0);

                w = w * wn;
            }
        }
    }

    if (inv == -1) {

            a[i] /= n;
        }
    }
}
